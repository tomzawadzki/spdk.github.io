105/**
106 * Get the value of the item with name 'key' in the section.
107 *
108 * If key appears multiple times, idx1 will control which version to retrieve.
109 * Indices will start from the top of the configuration file at 0 and increment
110 * by one for each new appearance. If the configuration key contains multiple
111 * whitespace delimited values, idx2 controls which value is returned. The index
112 * begins at 0.
113 *
114 *
115 * \param sp The section of the configuration.
116 * \param key Name of item.
117 * \param idx1 The index into the item list for the key.
118 * \param idx2 The index into the value list for the item.
119 *
120 * \return the requested value on success or NULL otherwise.
121 */
122char *spdk_conf_section_get_nmval(struct spdk_conf_section *sp, const char *key,
123 int idx1, int idx2);
124
125/**
126 * Get the first value of the item with name 'key' in the section.
127 *
128 * \param sp The section of the configuration.
129 * \param key Name of item.
130 * \param idx The index into the value list for the item.
131 *
132 * \return the requested value on success or NULL otherwise.
133 */
134char *spdk_conf_section_get_nval(struct spdk_conf_section *sp, const char *key, int idx);
135
136/**
137 * Get the first value of the first item with name 'key' in the section.
138 *
139 * \param sp The section of the configuration.
140 * \param key Name of item.
141 *
142 * \return the requested value on success or NULL otherwise.
143 */
144char *spdk_conf_section_get_val(struct spdk_conf_section *sp, const char *key);
145
146/**
147 * Get the first value of the first item with name 'key' in the section.
148 *
149 * \param sp The section of the configuration.
150 * \param key Name of item.
151 *
152 * \return the requested value on success or NULL otherwise.
153 */
154int spdk_conf_section_get_intval(struct spdk_conf_section *sp, const char *key);
155
156/**
157 * Get the bool value of the item with name 'key' in the section.
158 *
159 * This is used to check whether the service is enabled.
160 *
161 * \param sp The section of the configuration.
162 * \param key Name of item.
163 * \param default_val Default value.
164 *
165 * \return true if matching 'Yes/Y/True', false if matching 'No/N/False', default value otherwise.
166 */
167bool spdk_conf_section_get_boolval(struct spdk_conf_section *sp, const char *key, bool default_val);
