49struct spdk_keyring_module {
50 /** Name of the module */
51 const char *name;
52
53 /** Initializes a module, called during keyring's initialization */
54 int (*init)(void);
55 /** Clean up resources allocated by a module. Called during keyring's cleanup */
56 void (*cleanup)(void);
57 /** Write module configuration to JSON */
58 void (*write_config)(struct spdk_json_write_ctx *w);
59 /**
60 * Probe if a key with a specified name is available. If it is, the module should add it to
61 * the keyring and return zero. Otherwise, -ENOKEY should be returned.
62 */
63 int (*probe_key)(const char *name);
64 /** Add a key to the keyring */
65 int (*add_key)(struct spdk_key *key, void *ctx);
66 /** Remove a key from the keyring */
67 void (*remove_key)(struct spdk_key *key);
68 /** Get keying material from a key */
69 int (*get_key)(struct spdk_key *key, void *buf, int len);
70 /** Get the size of the context associated with a key */
71 size_t (*get_ctx_size)(void);
72 /**
73 * Dump information about a key to JSON. This callback should never dump keying material
74 * itself, only non-sensitive properties of a key must be dumped.
75 */
76 void (*dump_info)(struct spdk_key *key, struct spdk_json_write_ctx *w);
77
78 TAILQ_ENTRY(spdk_keyring_module) tailq;
79};
