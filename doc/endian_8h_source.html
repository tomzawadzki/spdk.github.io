20static inline uint16_t
21from_be16(const void *ptr)
22{
23 const uint8_t *tmp = (const uint8_t *)ptr;
24 return (((uint16_t)tmp[0] << 8) | tmp[1]);
25}

27static inline void
28to_be16(void *out, uint16_t in)
29{
30 uint8_t *tmp = (uint8_t *)out;
31 tmp[0] = (in >> 8) & 0xFF;
32 tmp[1] = in & 0xFF;
33}
