39/**
40 * Get the string value for a given sysfs attribute path
41 *
42 * When successful, the returned string will be null-terminated, without
43 * a trailing newline.
44 *
45 * \param attribute output parameter for contents of the attribute; caller must
46 * free() the buffer pointed to by attribute at some
47 * point after a successful call
48 * \param path_format format string for constructing patch to sysfs file
49 *
50 * \return 0 on success
51 * negative errno if unable to read the attribute
52 */
53int spdk_read_sysfs_attribute(char **attribute, const char *path_format, ...)
54__attribute__((format(printf, 2, 3)));
