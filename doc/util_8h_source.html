54/**
55 * Macro to align a value to a given power-of-two. The resultant value
56 * will be of the same type as the first parameter, and will be no
57 * bigger than the first parameter. Second parameter must be a
58 * power-of-two value.

60#define SPDK_ALIGN_FLOOR(val, align) \

96/**
97 * Generate a 64-bit pseudo-random number using xorshift algorithm.
98 *
99 * \param state the current seed value.
100 * \return a new pseudo-random 64-bit number.
101 */
102static inline uint64_t
103spdk_rand_xorshift64(uint64_t *state)
104{
105 uint64_t x = *state;
106
107 x ^= x << 13;
108 x ^= x >> 7;
109 x ^= x << 17;
110
111 *state = x;
112 return x;
113}

115/**
116 * Generate a non-zero initial seed for xorshift64.
117 *
118 * \return a random 64-bit seed value(non-zero).
119 */
120static inline uint64_t
121spdk_rand_xorshift64_seed(void)
122{
123 uint64_t seed = ((uint64_t)rand() << 32) | rand();
124
125 /* Avoid zero seed */
126 if (seed == 0) {
127 seed = 1;
128 }
129
130 return seed;
131}

325/**
326 * Scan build is really pessimistic and assumes that mempool functions can
327 * dequeue NULL buffers even if they return success. This is obviously a false
328 * positive, but the mempool dequeue can be done in a DPDK inline function that
329 * we can't decorate with usual assert(buf != NULL). Instead, we'll
330 * preinitialize the dequeued buffer array with some dummy objects.
331 */
332#define SPDK_CLANG_ANALYZER_PREINIT_PTR_ARRAY(arr, arr_size, buf_size) \
333 do { \
334 static char dummy_buf[buf_size]; \
335 int i; \
336 for (i = 0; i < arr_size; i++) { \
337 arr[i] = (void *)dummy_buf; \
338 } \
339 } while (0)

341/**
342 * Add two sequence numbers s1 and s2
343 *
344 * \param s1 First sequence number
345 * \param s2 Second sequence number
346 *
347 * \return Sum of s1 and s2 based on serial number arithmetic.
348 */
349static inline uint32_t
350spdk_sn32_add(uint32_t s1, uint32_t s2)
351{
352 return (uint32_t)(s1 + s2);
353}
354
355#define SPDK_SN32_CMPMAX (1U << (32 - 1))
356
357/**
358 * Compare if sequence number s1 is less than s2.
359 *
360 * \param s1 First sequence number
361 * \param s2 Second sequence number
362 *
363 * \return true if s1 is less than s2, or false otherwise.
364 */
365static inline bool
366spdk_sn32_lt(uint32_t s1, uint32_t s2)
367{
368 return (s1 != s2) &&
369 ((s1 < s2 && s2 - s1 < SPDK_SN32_CMPMAX) ||
370 (s1 > s2 && s1 - s2 > SPDK_SN32_CMPMAX));
371}

373/**
374 * Compare if sequence number s1 is greater than s2.
375 *
376 * \param s1 First sequence number
377 * \param s2 Second sequence number
378 *
379 * \return true if s1 is greater than s2, or false otherwise.
380 */
381static inline bool
382spdk_sn32_gt(uint32_t s1, uint32_t s2)
383{
384 return (s1 != s2) &&
385 ((s1 < s2 && s2 - s1 > SPDK_SN32_CMPMAX) ||
386 (s1 > s2 && s1 - s2 < SPDK_SN32_CMPMAX));

389/**
390 * Copies the value (unsigned char)ch into each of the first \b count characters of the object pointed to by \b data
391 * \b data_size is used to check that filling \b count bytes won't lead to buffer overflow
392 *
393 * \param data Buffer to fill
394 * \param data_size Size of the buffer
395 * \param ch Fill byte
396 * \param count Number of bytes to fill
397 */
398static inline void
399spdk_memset_s(void *data, size_t data_size, int ch, size_t count)
400{
401#ifdef __STDC_LIB_EXT1__
402 /* memset_s was introduced as an optional feature in C11 */
403 memset_s(data, data_size, ch, count);
404#else
405 size_t i;
406 volatile unsigned char *buf = (volatile unsigned char *)data;
407
408 if (!buf) {
409 return;
410 }
411 if (count > data_size) {
412 count = data_size;
413 }
414
415 for (i = 0; i < count; i++) {
416 buf[i] = (unsigned char)ch;
417 }
418#endif
419}
