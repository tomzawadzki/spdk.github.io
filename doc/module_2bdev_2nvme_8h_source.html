115/**
116 * Connect to the NVMe controller and populate namespaces as bdevs.
117 *
118 * \param trid Transport ID for nvme controller.
119 * \param base_name Base name for the nvme subsystem.
120 * \param names Pointer to string array to get bdev names.
121 * \param count Maximum count of the string array 'names'. Restricts the length
122 * of 'names' array only, not the count of bdevs created.
123 * \param cb_fn Callback function to be called after all the bdevs are created

125 * \param cb_ctx Context to pass to cb_fn.

128 * \return 0 on success, negative errno on failure.

139/**
140 * Delete the specified NVMe controller, or one of its paths.
141 *
142 * NOTE: When path_id is specified and it is the only path_id associated with NVMe controller
143 * the path is removed and the NVMe controller gets deleted. (Optional) callback
144 * function gets executed on delete complete in caller's thread. When the (optional)
145 * callback is not provided, the control is returned back at the time delete is initiated,
146 * not when it is completed. When NVMe controller deletion is already in progress state,
147 * this function returns success.
148 *
149 * \param name NVMe controller name.
150 * \param path_id The specified path to remove (optional).
151 * \param delete_cb Callback function on delete complete (optional).
152 * \param cb_ctx Context passed to callback (optional).
153 * \return zero on success,
154 * -EINVAL on wrong parameters or
155 * -ENODEV if controller is not found or
156 * -ENOMEM on no memory
157 */
158int spdk_bdev_nvme_delete(const char *name, const struct spdk_nvme_path_id *path_id,
159 spdk_bdev_nvme_delete_cb delete_cb, void *cb_ctx);

161/**
162 * Set multipath policy of the NVMe bdev.
163 *
164 * Subsequent calls for the same bdev must wait until the current execution finishes.
165 *
166 * \param name NVMe bdev name.
167 * \param policy Multipath policy (active-passive or active-active).
168 * \param selector Multipath selector (round_robin, queue_depth).
169 * \param rr_min_io Number of IO to route to a path before switching to another for round-robin.
170 * \param cb_fn Function to be called back after completion.
171 * \param cb_arg Argument passed to the callback function.
172 */
173void spdk_bdev_nvme_set_multipath_policy(const char *name,
174 enum spdk_bdev_nvme_multipath_policy policy,

176 uint32_t rr_min_io,
