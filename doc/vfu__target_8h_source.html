46/**
47 * Vfio-user PCI device sparse MMAP region.
48 *
49 * The sparse mmap allows finer granularity of specifying areas
50 * within a PCI region with mmap support.
51 */
52struct spdk_vfu_sparse_mmap {
53 /**
54 * Sparse mmap region offset, starts from 0 of current PCI region.
55 */
56 uint64_t offset;
57
58 /**
59 * Sparse mmap region length.
60 */
61 uint64_t len;
62};

64#define SPDK_VFU_MAXIMUM_SPARSE_MMAP_REGIONS 8

66/**
67 * Callback for vfio-user PCI region access.
68 *
69 * \param vfu_ctx Opaque value of the PCI endpoint, it's created by libvfio-user.
70 * \param buf data buffer to R/W.
71 * \param count R/W size, the value could be 1,2,4,8.
72 * \param pos offset from PCI region.
73 * \param is_write true if access is WRITE.
74 *
75 * \return count on success, negative errno on failure.
76 */
77typedef ssize_t (*spdk_vfu_access_cb)(vfu_ctx_t *vfu_ctx, char *buf, size_t count, loff_t pos,
78 bool is_write);

122/**
123 * Vfio-user PCI device information.
124 *
125 * vfio-user target uses this data structure to get all the information
126 * from backend emulated device module.
127 */
128struct spdk_vfu_pci_device {
129 struct {
130 /* Vendor ID */
131 uint16_t vid;
132 /* Device ID */
133 uint16_t did;
134 /* Subsystem Vendor ID */
135 uint16_t ssvid;
136 /* Subsystem ID */
137 uint16_t ssid;
138 } id;
139
140 struct {
141 /* Base Class Code */
142 uint8_t bcc;
143 /* Sub Class code */
144 uint8_t scc;
145 /* Programming Interface */
146 uint8_t pi;
147 } class;
148
149 /* Standard Power Management capabilities */
150 struct pmcap pmcap;
151 /* Standard PCI Express Capability ID */
152 struct pxcap pxcap;
153 /* Standard MSI-X Capability */
154 struct msixcap msixcap;
155 /* Number of vendor specific capabilities */
156 uint16_t nr_vendor_caps;
157
158 /* Legacy interrupt pin number */
159 uint16_t intr_ipin;
160 /* Number of legacy interrupts */
161 uint32_t nr_int_irqs;
162 /* Number of MSIX interrupts */
163 uint32_t nr_msix_irqs;
164
165 /* PCI regions */
166 struct spdk_vfu_pci_region regions[VFU_PCI_DEV_NUM_REGIONS];
167};
