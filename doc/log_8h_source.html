149#define SPDK_NOTICELOG(...) \
150 spdk_log(SPDK_LOG_NOTICE, __FILE__, __LINE__, __func__, __VA_ARGS__)
151#define SPDK_WARNLOG(...) \
152 spdk_log(SPDK_LOG_WARN, __FILE__, __LINE__, __func__, __VA_ARGS__)
153#define SPDK_ERRLOG(...) \
154 spdk_log(SPDK_LOG_ERROR, __FILE__, __LINE__, __func__, __VA_ARGS__)
155#define SPDK_PRINTF(...) \
156 spdk_log(SPDK_LOG_NOTICE, NULL, -1, NULL, __VA_ARGS__)
157#define SPDK_INFOLOG(flag, ...) \
158 do { \
159 extern struct spdk_log_flag SPDK_LOG_##flag; \
160 if (SPDK_LOG_##flag.enabled) { \
161 spdk_log(SPDK_LOG_INFO, __FILE__, __LINE__, __func__, __VA_ARGS__); \
162 } \
163 } while (0)
164
165#define SPDK_ERRLOG_RATELIMIT(...) \
166 do { \
167 static uint64_t last_tsc = 0; \
168 static uint64_t squashed = 0; \
169 uint64_t tsc = spdk_get_ticks(); \
170 if (tsc > last_tsc + spdk_get_ticks_hz()) { \
171 last_tsc = tsc; \
172 SPDK_ERRLOG(__VA_ARGS__); \
173 if (squashed > 0) { \
174 SPDK_ERRLOG("(same message squashed %" PRIu64 " times)\n", \
175 squashed); \
176 squashed = 0; \
177 } \
178 } else { \
179 squashed++; \
180 } \
181 } while (0)
182
183#ifdef DEBUG
184#define SPDK_DEBUGLOG(flag, ...) \
185 do { \
186 extern struct spdk_log_flag SPDK_LOG_##flag; \
187 if (SPDK_LOG_##flag.enabled) { \
188 spdk_log(SPDK_LOG_DEBUG, __FILE__, __LINE__, __func__, __VA_ARGS__); \
189 } \
190 } while (0)
191
192#define SPDK_LOGDUMP(flag, label, buf, len) \
193 do { \
194 extern struct spdk_log_flag SPDK_LOG_##flag; \
195 if (SPDK_LOG_##flag.enabled) { \
196 spdk_log_dump(stderr, (label), (buf), (len)); \
197 } \
198 } while (0)

346/**
347 * Register a deprecation. Most consumers will use SPDK_LOG_DEPRECATION_REGISTER() instead.
348 *
349 * \param tag A unique string that will appear in each log message and should appear in
350 * documentation.
351 * \param description A descriptive string that will also be logged.
352 * \param rate_limit_seconds If non-zero, log messages related to this deprecation will appear no
353 * more frequently than this interval.
354 * \param remove_release The release when the deprecated support will be removed.
355 * \param reg Pointer to storage for newly allocated deprecation handle.
356 * \return 0 on success or negative errno on failure.
357 */
358int spdk_log_deprecation_register(const char *tag, const char *description,
359 const char *remove_release, uint32_t rate_limit_seconds,
360 struct spdk_deprecation **reg);
361
362#define SPDK_LOG_DEPRECATION_REGISTER(tag, desc, release, rate) \
363 static struct spdk_deprecation *_deprecated_##tag; \
364 static void __attribute__((constructor)) _spdk_deprecation_register_##tag(void) \
365 { \
366 int rc; \
367 rc = spdk_log_deprecation_register(#tag, desc, release, rate, &_deprecated_##tag); \
368 (void)rc; \
369 assert(rc == 0); \
370 }
