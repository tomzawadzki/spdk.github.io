112/* Give SPDK_RPC_REGISTER a higher execution priority than
113 * SPDK_RPC_REGISTER_ALIAS_DEPRECATED to ensure all of the RPCs are registered
114 * before we try registering any aliases. Some older versions of clang may
115 * otherwise execute the constructors in a different order than
116 * defined in the source file (see issue #892).
117 */
118#define SPDK_RPC_REGISTER(method, func, state_mask) \
119static void __attribute__((constructor(1000))) rpc_register_##func(void) \
120{ \
121 spdk_rpc_register_method(method, func, state_mask); \
122}
123
124#define SPDK_RPC_REGISTER_ALIAS_DEPRECATED(method, alias) \
125static void __attribute__((constructor(1001))) rpc_register_##alias(void) \
126{ \
127 spdk_rpc_register_alias_deprecated(#method, #alias); \
128}
