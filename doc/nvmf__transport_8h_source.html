693static inline enum spdk_nvme_data_transfer
694spdk_nvmf_req_get_xfer(struct spdk_nvmf_request *req) {
695 enum spdk_nvme_data_transfer xfer;
696 struct spdk_nvme_cmd *cmd = &req->cmd->nvme_cmd;
697 struct spdk_nvme_sgl_descriptor *sgl = &cmd->dptr.sgl1;
698
699 /* Figure out data transfer direction */
700 if (cmd->opc == SPDK_NVME_OPC_FABRIC)
701 {
702 xfer = spdk_nvme_opc_get_data_transfer(req->cmd->nvmf_cmd.fctype);
703 } else
704 {
705 xfer = spdk_nvme_opc_get_data_transfer(cmd->opc);
706 }
707
708 if (xfer == SPDK_NVME_DATA_NONE)
709 {
710 return xfer;
711 }
712
713 /* Even for commands that may transfer data, they could have specified 0 length.
714 * We want those to show up with xfer SPDK_NVME_DATA_NONE.
715 */
716 switch (sgl->generic.type)
717 {
718 case SPDK_NVME_SGL_TYPE_DATA_BLOCK:
719 case SPDK_NVME_SGL_TYPE_BIT_BUCKET:
720 case SPDK_NVME_SGL_TYPE_SEGMENT:
721 case SPDK_NVME_SGL_TYPE_LAST_SEGMENT:
722 case SPDK_NVME_SGL_TYPE_TRANSPORT_DATA_BLOCK:
723 if (sgl->unkeyed.length == 0) {
724 xfer = SPDK_NVME_DATA_NONE;
725 }
726 break;
727 case SPDK_NVME_SGL_TYPE_KEYED_DATA_BLOCK:
728 if (sgl->keyed.length == 0) {
729 xfer = SPDK_NVME_DATA_NONE;
730 }
731 break;
732 }
733
734 return xfer;
735}
