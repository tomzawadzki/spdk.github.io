177struct spdk_nvme_intel_marketing_description_page {
178 uint8_t marketing_product[512];
179 /* Spec says this log page will only write 512 bytes, but there are some older FW
180 * versions that accidentally write 516 instead. So just pad this out to 4096 bytes
181 * to make sure users of this structure never end up overwriting unintended parts of

185};
