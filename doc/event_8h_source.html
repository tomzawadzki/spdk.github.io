215/**
216 * Start the framework.
217 *
218 * Before calling this function, opts must be initialized by
219 * spdk_app_opts_init(). Once started, the framework will call start_fn on
220 * an spdk_thread running on the current system thread with the
221 * argument provided.
222 *
223 * If opts->delay_subsystem_init is set
224 * (e.g. through --wait-for-rpc flag in spdk_app_parse_args())
225 * this function will only start a limited RPC server accepting
226 * only a few RPC commands - mostly related to pre-initialization.
227 * With this option, the framework won't be started and start_fn
228 * won't be called until the user sends an `rpc_framework_start_init`
229 * RPC command, which marks the pre-initialization complete and
230 * allows start_fn to be finally called.
231 *
232 * This call will block until spdk_app_stop() is called. If an error
233 * condition occurs during the initialization code within spdk_app_start(),
234 * this function will immediately return before invoking start_fn.
235 *
236 * \param opts_user Initialization options used for this application. It should not be
237 * NULL. And the opts_size value inside the opts structure should not be zero.
238 * \param start_fn Entry point that will execute on an internally created thread
239 * once the framework has been started.
240 * \param ctx Argument passed to function start_fn.
241 *
242 * \return 0 on success or non-zero on failure.
243 */
244int spdk_app_start(struct spdk_app_opts *opts_user, spdk_msg_fn start_fn,
245 void *ctx);

253 * Start shutting down the framework.
254 *
255 * Typically this function is not called directly, and the shutdown process is
