25/**
26 * Structure with optional parameters for the JSON-RPC server initialization.

28struct spdk_rpc_opts {
29 /* Size of this structure in bytes. */

32 * A JSON-RPC log file pointer. The default value is NULL and used

44/**
45 * Create SPDK JSON-RPC server listening at provided address and start polling it for connections.
46 *
47 * This function should be called from the SPDK app thread.
48 *
49 * The RPC server is optional and is independent of subsystem initialization.
50 * The RPC server can be started and stopped at any time.
51 *
52 * \param listen_addr Path to a unix domain socket to listen on
53 * \param opts Options for JSON-RPC server initialization. If NULL, default values are used.
54 *
55 * \return Negated errno on failure. 0 on success.
56 */

76typedef void (*spdk_subsystem_init_fn)(int rc, void *ctx);

78/**
79 * Begin the initialization process for all SPDK subsystems.
80 *
81 * This function should be called from the SPDK app thread.
82 *
83 * SPDK is divided into subsystems at a macro-level and each subsystem automatically registers
84 * itself with this library at start up using a C constructor. Further, each subsystem can declare
85 * other subsystems that it depends on. Calling this function will correctly initialize all
86 * subsystems that are present, in the required order.
87 *
88 * \param cb_fn Function called when the process is complete.
89 * \param cb_arg User context passed to cb_fn.
90 */
91void spdk_subsystem_init(spdk_subsystem_init_fn cb_fn, void *cb_arg);
92
93/**
94 * Loads RPC configuration from provided JSON for current RPC state.
95 *
96 * This function should be called from the SPDK app thread.
97 *
98 * The function will automatically start a JSON RPC server for configuration purposes and then stop
99 * it. JSON data will be copied, so parsing will not disturb the original memory.
100 *
101 * \param json Raw JSON data.
102 * \param json_size Size of JSON data.
103 * \param cb_fn Function called when the process is complete.
104 * \param cb_arg User context passed to cb_fn.
105 * \param stop_on_error Whether to stop initialization if one of the JSON RPCs fails.
106 */
107void spdk_subsystem_load_config(void *json, ssize_t json_size, spdk_subsystem_init_fn cb_fn,
108 void *cb_arg, bool stop_on_error);
