22/**
23 * File descriptor type. The event handler may have extra checks and can do extra
24 * processing based on this.
25 */
26enum spdk_fd_type {
27 SPDK_FD_TYPE_DEFAULT = 0x0,
28 /**
29 * Event file descriptors. Once an event is generated on these file descriptors, event
30 * handler will perform a read operation on it to reset its internal eventfd object
31 * counter value to 0.
32 */
33 SPDK_FD_TYPE_EVENTFD = 0x1,
34};
35
36struct spdk_event_handler_opts {
37 /**
38 * The size of spdk_event_handler_opts according to the caller of this library is used for
39 * ABI compatibility. The library uses this field to know how many fields in this structure
40 * are valid. And the library will populate any remaining fields with default values.
41 * New added fields should be put at the end of the struct.
42 */
43 size_t opts_size;
44
45 /** Event notification types */
46 uint32_t events;
47
48 /** fd type \ref spdk_fd_type */
49 uint32_t fd_type;
50};
51SPDK_STATIC_ASSERT(sizeof(struct spdk_event_handler_opts) == 16, "Incorrect size");

53/**
54 * Callback function registered for the event source file descriptor.
55 *
56 * \param ctx Context passed as arg to spdk_fd_group_add().
57 *
58 * \return 0 to indicate that event notification took place but no events were found;
59 * positive to indicate that event notification took place and some events were processed;
60 * negative if no event information is provided.
61 */
62typedef int (*spdk_fd_fn)(void *ctx);

64/**
65 * A file descriptor group of event sources which gather the events to an epoll instance.

67 * Taking "fgrp" as short name for file descriptor group of event sources.

69struct spdk_fd_group;

98/**
99 * Wait for new events generated inside fgrp, and process them with their
100 * registered spdk_fd_fn.
101 *
102 * \param fgrp The fgrp to wait and process.
103 * \param timeout Specifies the number of milliseconds that will block.
104 * -1 causes indefinitely blocking; 0 causes immediately return.
105 *
106 * \return the number of events processed on success, negated errno on failure.
107 */
108int spdk_fd_group_wait(struct spdk_fd_group *fgrp, int timeout);
109
110/**
111 * Return the internal epoll_fd of specific fd_group
112 *
113 * \param fgrp The pointer of specified fgrp.
114 *
115 * \return The epoll_fd of specific fgrp.
116 */
117int spdk_fd_group_get_fd(struct spdk_fd_group *fgrp);

119/**
120 * Nest the child fd_group in the parent fd_group. After this operation
121 * completes, calling spdk_fd_group_wait() on the parent will include events
122 * from the child.
123 *
124 * \param parent The parent fd_group.
125 * \param child The child fd_group.
126 *
127 * \return 0 on success. Negated errno on failure. However, on all errno values other
128 * than -ENOTRECOVERABLE, the operation has not changed the state of the fd_group.
129 */
130int spdk_fd_group_nest(struct spdk_fd_group *parent, struct spdk_fd_group *child);
131
132/**
133 * Remove the nested child from the parent.
134 *
135 * \param parent The parent fd_group.
136 * \param child The child fd_group.
137 *
138 * \return 0 on success. Negated errno on failure. However, on all errno values other
139 * than -ENOTRECOVERABLE, the operation has not changed the state of the fd_group.
140 */
141int spdk_fd_group_unnest(struct spdk_fd_group *parent, struct spdk_fd_group *child);

143/**
144 * Register SPDK_INTERRUPT_EVENT_IN event source to specified fgrp.
145 *
146 * Use spdk_fd_group_add_for_events() for other event types.
147 *
148 * \param fgrp The fgrp registered to.
149 * \param efd File descriptor of the event source.
150 * \param fn Called each time there are events in event source.
151 * \param arg Function argument for fn.
152 * \param name Name of the event source.
153 *
154 * \return 0 on success, negated errno on failure.
155 */
156int spdk_fd_group_add(struct spdk_fd_group *fgrp, int efd,
157 spdk_fd_fn fn, void *arg, const char *name);
158
159/**
160 * Register one event source to specified fgrp with specific event types.
161 *
162 * Event types argument is a bit mask composed by ORing together
163 * enum spdk_interrupt_event_types values.
164 *
165 * \param fgrp The fgrp registered to.
166 * \param efd File descriptor of the event source.
167 * \param events Event notification types.
168 * \param fn Called each time there are events in event source.
169 * \param arg Function argument for fn.
170 * \param name Name of the event source.
171 *
172 * \return 0 on success, negated errno on failure.
173 */
174int spdk_fd_group_add_for_events(struct spdk_fd_group *fgrp, int efd, uint32_t events,
175 spdk_fd_fn fn, void *arg, const char *name);
176
177/**
178 * Register one event type stated in spdk_event_handler_opts agrument to the specified fgrp.
179 *
180 * spdk_event_handler_opts argument consists of event which is a bit mask composed by ORing
181 * together enum spdk_interrupt_event_types values. It also consists of fd_type, which can be
182 * used by event handler to perform extra checks during the spdk_fd_group_wait call.
183 *
184 * \param fgrp The fgrp registered to.
185 * \param efd File descriptor of the event source.
186 * \param fn Called each time there are events in event source.
187 * \param arg Function argument for fn.
188 * \param name Name of the event source.
189 * \param opts Extended event handler option.
190 *
191 * \return 0 if success or -errno if failed
192 */
193int spdk_fd_group_add_ext(struct spdk_fd_group *fgrp, int efd, spdk_fd_fn fn, void *arg,
194 const char *name, struct spdk_event_handler_opts *opts);

218/**
219 * Change the event notification types associated with the event source.
220 *
221 * Modules like nbd, need this api to add EPOLLOUT when having data to send, and remove EPOLLOUT if no data to send.

223 * \param fgrp The fgrp registered to.

227 * \return 0 on success, negated errno on failure.

229int spdk_fd_group_event_modify(struct spdk_fd_group *fgrp,

238/**
239 * Copies the epoll(7) event that caused a callback function to execute.
240 * This function can only be called by the callback function, doing otherwise
241 * results in undefined behavior.
242 *
243 * \param event pointer to an epoll(7) event to copy to.
244 * \return 0 on success, negated errno on failure.

246int spdk_fd_group_get_epoll_event(struct epoll_event *event);

248typedef int (*spdk_fd_group_wrapper_fn)(void *wrapper_ctx, spdk_fd_fn cb_fn, void *cb_ctx);

250/**
251 * Set a wrapper function to be called when an epoll(7) event is received. The callback associated
252 * with that event is passed to the wrapper, which is responsible for executing it. Only one
253 * wrapper can be assigned to an fd_group at a time.
254 *
255 * \param fgrp fd group.
256 * \param cb_fn Wrapper callback.
257 * \param cb_ctx Wrapper callback's context.
258 *
259 * \return 0 on success, negative errno otherwise.
260 */
261int spdk_fd_group_set_wrapper(struct spdk_fd_group *fgrp, spdk_fd_group_wrapper_fn cb_fn,
262 void *cb_ctx);
