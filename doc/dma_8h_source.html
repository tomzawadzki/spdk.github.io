26enum spdk_dma_device_type {
27 /** RDMA devices are capable of performing DMA operations on memory domains using the standard
28 * RDMA model (protection domain, remote key, address). */
29 SPDK_DMA_DEVICE_TYPE_RDMA,
30 /** DMA devices are capable of performing DMA operations on memory domains using physical or
31 * I/O virtual addresses. */
32 SPDK_DMA_DEVICE_TYPE_DMA,
33 /** Virtual memory domain representing memory being transformed by accel framework */
34 SPDK_DMA_DEVICE_TYPE_ACCEL,
35 /**
36 * Start of the range of vendor-specific DMA device types
37 */
38 SPDK_DMA_DEVICE_VENDOR_SPECIFIC_TYPE_START = 1000,
39 /**
40 * End of the range of vendor-specific DMA device types
41 */
42 SPDK_DMA_DEVICE_VENDOR_SPECIFIC_TYPE_END = SPDK_DMA_DEVICE_VENDOR_SPECIFIC_TYPE_START + 999
43};

55/**

57 * Implementation of this function must call \b cpl_cb only when it returns 0. All other return codes mean failure.

213/**
214 * Creates a new memory domain of the specified type.
215 *
216 * Translation functions can be provided to translate addresses from one memory domain to another.
217 * If the two domains both use the same addressing scheme for, then this translation does nothing.
218 * However, it is possible that the two memory domains may address the same physical memory
219 * differently, so this translation step is required.
220 *
221 * \param domain Double pointer to memory domain to be allocated by this function
222 * \param type Type of the DMA device which can access this memory domain
223 * \param ctx Optional memory domain context to be copied by this function. Later \b ctx can be
224 * retrieved using \ref spdk_memory_domain_get_context function
225 * \param id String identifier representing the DMA device that can access this memory domain.
226 * \return 0 on success, negated errno on failure
227 */

385/**
386 * Translate data located in \b src_domain space at address \b addr with size \b len into an equivalent
387 * description of memory in dst_domain.
388 *
389 * This function calls \b src_domain translation callback, the callback needs to be set using \ref
390 * spdk_memory_domain_set_translation function.
391 * No data is moved during this operation. Both src_domain and dst_domain must describe the same physical memory,
392 * just from the point of view of two different memory domain. This is a translation of the description of the memory only.
393 * Result of translation is stored in \b result, its content depends on the type of \b dst_domain.
394 *
395 * \param src_domain Memory domain in which address space data buffer is located
396 * \param src_domain_ctx User defined context
397 * \param dst_domain Memory domain in which memory space data buffer should be translated
398 * \param dst_domain_ctx Ancillary data for dst_domain
399 * \param addr Address in \b src_domain memory space
400 * \param len Length of the data
401 * \param result Translation result. The content of the translation result is only valid if this
402 * function returns 0.
403 * \return 0 on success, negated errno on failure.
404 */
405int spdk_memory_domain_translate_data(struct spdk_memory_domain *src_domain, void *src_domain_ctx,
406 struct spdk_memory_domain *dst_domain, struct spdk_memory_domain_translation_ctx *dst_domain_ctx,
407 void *addr, size_t len, struct spdk_memory_domain_translation_result *result);
