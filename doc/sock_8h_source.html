26/**
27 * Anywhere this struct is used, an iovec array is assumed to
28 * immediately follow the last member in memory, without any
29 * padding.
30 *
31 * A simpler implementation would be to place a 0-length array
32 * of struct iovec at the end of this request. However, embedding
33 * a structure that ends with a variable length array inside of
34 * another structure is a GNU C extension and not standard.
35 */
36struct spdk_sock_request {
37 /* When the request is completed, this callback will be called.
38 * On success, err will be:
39 * - for writes: 0,
40 * - for reads: number of bytes read.
41 * On failure: negative errno value.
42 */
43 void (*cb_fn)(void *cb_arg, int err);
44 void *cb_arg;
45
46 /* These fields are used by the socket layer and should not be modified. */
47 struct __sock_request_internal {
48 TAILQ_ENTRY(spdk_sock_request) link;
49
50 /**
51 * curr_list is only used in DEBUG mode, but we include it in
52 * release builds too to ensure ABI compatibility between debug
53 * and release builds.
54 */
55 void *curr_list;
56
57 uint32_t offset;
58
59 /* Last zero-copy sendmsg index. */
60 uint32_t zcopy_idx;
61
62 /* Indicate if the whole req or part of it is pending zerocopy completion. */
63 bool pending_zcopy;
64 } internal;
65
66 int iovcnt;
67 /* struct iovec iov[]; */
68};
69
70SPDK_STATIC_ASSERT(sizeof(struct spdk_sock_request) == 64, "Incorrect size.");

85/**

88 * A pointer to this structure is used by spdk_sock_impl_get_opts() and spdk_sock_impl_set_opts()

114 /**
115 * Enable or disable quick ACK. Used by posix and uring socket modules.

121 * Valid values in the enum spdk_placement_mode.
122 */

125 /**
126 * Enable or disable use of zero copy flow on send for server sockets. Used by posix and uring socket modules.

196/**

199 * A pointer to this structure will be used by spdk_sock_listen_ext() or spdk_sock_connect_ext() to
200 * allow the user to request non-default options on the socket.

202struct spdk_sock_opts {
203 /**
204 * The size of spdk_sock_opts according to the caller of this library is used for ABI

206 * structure are valid. And the library will populate any remaining fields with default values.

224 * Time in msec to wait ack until connection is closed forcefully.

231 /**
232 * Socket implementation options. If non-NULL, these will override those set by

234 * free it immediately after a spdk_sock_connect()/spdk_sock_listen() call.
235 */

237
238 /**
239 * Size of the impl_opts structure.

242
243 /**
244 * Source address. If NULL, any available address will be used. Only valid for connect().

248 /**
249 * Source port. If zero, a random ephemeral port will be used. Only valid for connect().

256 /**
257 * Time in msec to wait until connection is done (0 = no timeout).
