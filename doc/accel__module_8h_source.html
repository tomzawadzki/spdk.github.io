28enum spdk_accel_crypto_tweak_mode {
29 /* Tweak[127:0] = {64'b0, LBA[63:0]} */
30 SPDK_ACCEL_CRYPTO_TWEAK_MODE_SIMPLE_LBA,
31
32 /* Tweak[127:0] = {1’b0, ~LBA[62:0], LBA[63:0]} */
33 SPDK_ACCEL_CRYPTO_TWEAK_MODE_JOIN_NEG_LBA_WITH_LBA,
34
35 /* Tweak is derived from LBA that is internally incremented by 1 for every 512 bytes processed
36 * so initial lba = (BLOCK_SIZE_IN_BYTES / 512) * LBA
37 * Tweak[127:0] = {lba[127:0]} */
38 SPDK_ACCEL_CRYPTO_TWEAK_MODE_INCR_512_FULL_LBA,
39
40 /* Tweak is derived from LBA that is internally incremented by 1 for every 512 bytes processed
41 * so initial lba = (BLOCK_SIZE_IN_BYTES / 512) * LBA
42 * Tweak[127:0] = {lba[63:0], 64'b0} */
43 SPDK_ACCEL_CRYPTO_TWEAK_MODE_INCR_512_UPPER_LBA,
44};

168struct spdk_accel_module_if {
169 /** Name of the module. */
170 const char *name;
171
172 /**
173 * Priority of the module. It's used to select a module to execute an operation when
174 * multiple modules support it. Higher value means higher priority. Software module has a
175 * priority of `SPDK_ACCEL_SW_PRIORITY`. Of course, this value is only relevant when none
176 * of the modules have been explicitly assigned to execute a given operation via
177 * `spdk_accel_assign_opc()`.
178 */
179 int priority;
180
181 /**
182 * Initialization function for the module. Called by the application during startup.
183 *
184 * Return 0 on success or negative error code. If -ENODEV is returned - the module
185 * will not be used to handle any operation, but the error will not stop framework
186 * initialization.
187 *
188 * Modules are required to define this function.
189 */
190 int (*module_init)(void);
191
192 /**
193 * Finish function for the module. Called by the application before the application exits
194 * to perform any necessary cleanup.
195 *
196 * Modules are not required to define this function.
197 */
198 void (*module_fini)(void *ctx);
199
200 /** Write Acceleration module configuration into provided JSON context. */
201 void (*write_config_json)(struct spdk_json_write_ctx *w);
202
203 /** Returns the allocation size required for the modules to use for context. */
204 size_t (*get_ctx_size)(void);
205
206 /** Reports whether the module supports a given operation. */
207 bool (*supports_opcode)(enum spdk_accel_opcode);
208
209 /** Returns module's IO channel on the calling thread. */
210 struct spdk_io_channel *(*get_io_channel)(void);
211
212 /**
213 * Submit tasks to be executed by the module. Once a task execution is done, the module is
214 * required to complete it using `spdk_accel_task_complete()`. `ch` is the IO channel
215 * obtained by `get_io_channel()`.
216 */
217 int (*submit_tasks)(struct spdk_io_channel *ch, struct spdk_accel_task *accel_task);
218
219 /**
220 * Create crypto key function. Module is responsible to fill all necessary parameters in
221 * \b spdk_accel_crypto_key structure
222 */
223 int (*crypto_key_init)(struct spdk_accel_crypto_key *key);
224
225 /** Free any resources associated with `key` allocated during `crypto_key_init()`. */
226 void (*crypto_key_deinit)(struct spdk_accel_crypto_key *key);
227
228 /**
229 * Returns true if given tweak mode is supported. If module doesn't implement that function it shall support SIMPLE LBA mode.
230 */
231 bool (*crypto_supports_tweak_mode)(enum spdk_accel_crypto_tweak_mode tweak_mode);
232
233 /**
234 * Returns true if given pair (cipher, key size) is supported.
235 */
236 bool (*crypto_supports_cipher)(enum spdk_accel_cipher cipher, size_t key_size);
237
238 /**
239 * Return true if compresssion algo is supported, false otherwise.
240 */
241 bool (*compress_supports_algo)(enum spdk_accel_comp_algo algo);
242
243 /**
244 * Returns the lowest and highest levels of the specified algorithm.
245 */
246 int (*get_compress_level_range)(enum spdk_accel_comp_algo algo,
247 uint32_t *min_level, uint32_t *max_level);
248
249 /**
250 * Returns memory domains supported by the module. If NULL, the module does not support
251 * memory domains. The `domains` array can be NULL, in which case this function only
252 * returns the number of supported memory domains.
253 *
254 * \param domains Memory domain array.
255 * \param num_domains Size of the `domains` array.
256 *
257 * \return Number of supported memory domains.
258 */
259 int (*get_memory_domains)(struct spdk_memory_domain **domains, int num_domains);
260
261 /**
262 * Returns information/constraints for a given operation. If unimplemented, it is assumed
263 * that the module doesn't have any constraints to execute any operation.
264 */
265 int (*get_operation_info)(enum spdk_accel_opcode opcode,
266 const struct spdk_accel_operation_exec_ctx *ctx,
267 struct spdk_accel_opcode_info *info);
268
269 TAILQ_ENTRY(spdk_accel_module_if) tailq;
270};

288/**
289 * Platform driver responsible for executing tasks in a sequence. If no driver is selected, tasks
290 * are submitted to accel modules. All drivers are required to be aware of memory domains.
291 */
292struct spdk_accel_driver {
293 /** Name of the driver. */
294 const char *name;
295
296 /** Initializes the driver, called when accel initializes. Optional. */
297 int (*init)(void);
298
299 /** Performs cleanup on resources allocated by the driver. Optional. */
300 void (*fini)(void);
301
302 /**
303 * Executes a sequence of accel operations. The driver should notify accel about each
304 * completed task using `spdk_accel_task_complete()`. Once all tasks are completed or the
305 * driver cannot proceed with a given task (e.g. because it doesn't handle specific opcode),
306 * accel should be notified via `spdk_accel_sequence_continue()`. If there are tasks left
307 * in a sequence, the first will be submitted to a module, while the rest will be sent back
308 * to the driver. `spdk_accel_sequence_continue()` should only be called if this function
309 * succeeds (i.e. returns 0).
310 *
311 * \param ch IO channel obtained by `get_io_channel()`.
312 * \param seq Sequence of tasks to execute.
313 *
314 * \return 0 on success, negative errno on failure.
315 */
316 int (*execute_sequence)(struct spdk_io_channel *ch, struct spdk_accel_sequence *seq);
317
318 /** Returns IO channel that will be passed to `execute_sequence()`. */
319 struct spdk_io_channel *(*get_io_channel)(void);
320
321 /**
322 * Returns information/constraints for a given operation. If unimplemented, it is assumed
323 * that the driver doesn't have any constraints to execute any operation.
324 */
325 int (*get_operation_info)(enum spdk_accel_opcode opcode,
326 const struct spdk_accel_operation_exec_ctx *ctx,
327 struct spdk_accel_opcode_info *info);
328
329 TAILQ_ENTRY(spdk_accel_driver) tailq;
330};
