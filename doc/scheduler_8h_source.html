29/**
30 * Cores governor
31 * Implements core frequency control for schedulers. Functions from this structure
32 * are invoked from scheduling reactor.
33 */
34struct spdk_governor {
35 const char *name;
36
37 /**
38 * Get available frequencies of a given core.
39 *
40 * \param lcore_id Core ID.
41 * \param freqs The buffer array to save the frequencies.
42 * \param num Number of frequencies to get.
43 *
44 * \return The number of frequencies returned in freqs. 0 on error.
45 * 0 is returned if it could not get the frequencies or
46 * if the freqs array is too small to fit the returned frequencies.
47 */
48 uint32_t (*get_core_avail_freqs)(uint32_t lcore_id, uint32_t *freqs, uint32_t num);
49
50 /**
51 * Get current frequency of a given core.
52 *
53 * \param lcore_id Core number.
54 *
55 * \return Currently set core frequency.
56 */
57 uint32_t (*get_core_curr_freq)(uint32_t lcore_id);
58
59 /**
60 * Increase core frequency to next available one.
61 *
62 * \param lcore_id Core number.
63 *
64 * \return 1 on success, 0 already at max frequency, negative on error.
65 */
66 int (*core_freq_up)(uint32_t lcore_id);
67
68 /**
69 * Decrease core frequency to next available one.
70 *
71 * \param lcore_id Core number.
72 *
73 * \return 1 on success, 0 already at min frequency, negative on error.
74 */
75 int (*core_freq_down)(uint32_t lcore_id);
76
77 /**
78 * Set core frequency to maximum available.
79 *
80 * \param lcore_id Core number.
81 *
82 * \return 1 on success, 0 already at max frequency, negative on error.
83 */
84 int (*set_core_freq_max)(uint32_t lcore_id);
85
86 /**
87 * Set core frequency to minimum available.
88 *
89 * \param lcore_id Core number.
90 *
91 * \return 1 on success, 0 already at min frequency, negative on error.
92 */
93 int (*set_core_freq_min)(uint32_t lcore_id);
94
95 /**
96 * Get capabilities of a given core.
97 *
98 * \param lcore_id Core number.
99 * \param capabilities Structure to fill with capabilities data.
100 *
101 * \return 0 on success, negative on error.
102 */
103 int (*get_core_capabilities)(uint32_t lcore_id, struct spdk_governor_capabilities *capabilities);
104
105 /**
106 * Output governor-specific information to a JSON stream.
107 *
108 * The JSON write context will be initialized with an open object, so the governor
109 * should write a name followed by a JSON value (most likely another nested object).
110 */
111 int (*dump_info_json)(struct spdk_json_write_ctx *w);
112
113 /**
114 * Initialize a governor.
115 *
116 * \return 0 on success, non-zero on error.
117 */
118 int (*init)(void);
119
120 /**
121 * Deinitialize a governor.
122 */
123 void (*deinit)(void);
124
125 TAILQ_ENTRY(spdk_governor) link;
126};
127
128/**
129 * Set the current governor.
130 *
131 * Always deinitializes previously set governor.
132 * No governor will be set if name parameter is NULL.
133 * This function should be invoked on scheduling reactor.
134 *
135 * \param name Name of the governor to be used.
136 *
137 * \return 0 on success or non-zero on failure.
138 */
139int spdk_governor_set(const char *name);
140
141/**
142 * Get currently set governor.
143 *
144 * \return a pointer to spdk governor or NULL if none is set.
145 */
146struct spdk_governor *spdk_governor_get(void);
147
148/**
149 * Add the given governor to the list of registered governors.
150 * This function should be invoked by referencing the macro
151 * SPDK_GOVERNOR_REGISTER in the governor c file.
152 *
153 * \param governor Governor to be added.
154 *
155 * \return 0 on success or non-zero on failure.
156 */
157void spdk_governor_register(struct spdk_governor *governor);
158
159/**
160 * Macro used to register new governors.
161 */
162#define SPDK_GOVERNOR_REGISTER(governor) \
163 static void __attribute__((constructor)) _spdk_governor_register_ ## governor(void) \
164 { \
165 spdk_governor_register(&governor); \
166 }

168/**
169 * Structure representing thread used for scheduling.
170 */
171struct spdk_scheduler_thread_info {
172 uint32_t lcore;
173 uint64_t thread_id;
174 /* stats over a lifetime of a thread */
175 struct spdk_thread_stats total_stats;
176 /* stats during the last scheduling period */
177 struct spdk_thread_stats current_stats;
178};
179
180/**
181 * A list of cores and threads which is used for scheduling.
182 */
183struct spdk_scheduler_core_info {
184 /* stats over a lifetime of a core */
185 uint64_t total_idle_tsc;
186 uint64_t total_busy_tsc;
187 /* stats during the last scheduling period */
188 uint64_t current_idle_tsc;
189 uint64_t current_busy_tsc;
190
191 uint32_t lcore;
192 uint32_t threads_count;
193 bool interrupt_mode;
194 struct spdk_scheduler_thread_info *thread_infos;
195 bool isolated;
196};
197
198/**
199 * Thread scheduler.
200 * Functions from this structure are invoked from scheduling reactor.
201 */
202struct spdk_scheduler {
203 const char *name;
204
205 /**
206 * This function is called to initialize a scheduler.
207 *
208 * \return 0 on success or non-zero on failure.
209 */
210 int (*init)(void);
211
212 /**
213 * This function is called to deinitialize a scheduler.
214 */
215 void (*deinit)(void);
216
217 /**
218 * Function to balance threads across cores by modifying
219 * the value of their lcore field.
220 *
221 * \param core_info Structure describing cores and threads on them.
222 * \param count Size of the core_info array.
223 */
224 void (*balance)(struct spdk_scheduler_core_info *core_info, uint32_t count);
225
226 /**
227 * Function to set scheduler parameters like load_limit.
228 *
229 * \param opts Pointer to spdk_json_val struct containing values of parameters
230 * to be set in scheduler.
231 */
232 int (*set_opts)(const struct spdk_json_val *opts);
233
234 /**
235 * Function to get current scheduler parameters like load_limit.
236 *
237 * \param ctx Pointer to spdk_json_write_ctx struct to be filled with current parameters.
238 */
239 void (*get_opts)(struct spdk_json_write_ctx *ctx);
240
241 TAILQ_ENTRY(spdk_scheduler) link;
242};
243
244/**
245 * Change current scheduler. If another scheduler was used prior,
246 * it will be deinitialized. No scheduler will be set if name parameter
247 * is NULL.
248 * This function should be invoked from scheduling reactor.
249 *
250 * \param name Name of the scheduler to be used.
251 *
252 * \return 0 on success or non-zero on failure.
253 */
254int spdk_scheduler_set(const char *name);
255
256/**
257 * Get currently set scheduler.
258 *
259 * \return a pointer to spdk scheduler or NULL if none is set.
260 */
261struct spdk_scheduler *spdk_scheduler_get(void);
262
263/**
264 * Change current scheduling period.
265 * Setting period to 0 disables scheduling.
266 *
267 * \param period Period to set in microseconds.
268 */
269void spdk_scheduler_set_period(uint64_t period);
270
271/**
272 * Get scheduling period of currently set scheduler.
273 *
274 * \return Scheduling period in microseconds.
275 */
276uint64_t spdk_scheduler_get_period(void);
277
278/**
279 * Add the given scheduler to the list of registered schedulers.
280 * This function should be invoked by referencing the macro
281 * SPDK_SCHEDULER_REGISTER in the scheduler c file.
282 *
283 * \param scheduler Scheduler to be added.
284 */
285void spdk_scheduler_register(struct spdk_scheduler *scheduler);
286
287/**
288 * Get lcore of scheduling reactor.
289 *
290 * All scheduler operations are performed from the scheduling reactor.
291 *
292 * \return lcore of scheduling reactor
293 */
294uint32_t spdk_scheduler_get_scheduling_lcore(void);
295
296/**
297 * Set scheduling reactor.
298 *
299 * All scheduler operations are performed from the scheduling reactor.
300 *
301 * \param lcore lcore of scheduling reactor
302 */
303bool spdk_scheduler_set_scheduling_lcore(uint32_t lcore);
304
