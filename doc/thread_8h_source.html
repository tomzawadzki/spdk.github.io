533/**
534 * Run the msg callback on the given thread. If this happens to be the current
535 * thread, the callback is executed immediately; otherwise a message is sent to
536 * the thread, and it's run asynchronously.
537 *
538 * Errors are handled internally and are fatal. Calling code can skip checking the return
539 * value as it has been left only for compatibility.
540 *
541 * \param thread The target thread.
542 * \param fn This function will be called on the given thread.
543 * \param ctx This context will be passed to fn when called.
544 *
545 * \return 0 left for API compatibility
546 */
547static inline int
548spdk_thread_exec_msg(const struct spdk_thread *thread, spdk_msg_fn fn, void *ctx)
549{
550 assert(thread != NULL);
551
552 if (spdk_get_thread() == thread) {
553 fn(ctx);
554 return 0;
555 }
556
557 return spdk_thread_send_msg(thread, fn, ctx);
558}

1020/**
1021 * A spinlock augmented with safety checks for use with SPDK.
1022 *
1023 * SPDK code that uses spdk_spinlock runs from an SPDK thread, which itself is associated with a
1024 * pthread. There are typically many SPDK threads associated with each pthread. The SPDK application
1025 * may migrate SPDK threads between pthreads from time to time to balance the load on those threads.
1026 * Migration of SPDK threads only happens when the thread is off CPU, and as such it is only safe to
1027 * hold a lock so long as an SPDK thread stays on CPU.
1028 *
1029 * It is not safe to lock a spinlock, return from the event or poller, then unlock it at some later
1030 * time because:
1031 *
1032 * - Even though the SPDK thread may be the same, the SPDK thread may be running on different
1033 * pthreads during lock and unlock. A pthread spinlock may consider this to be an unlock by a
1034 * non-owner, which results in undefined behavior.
1035 * - A lock that is acquired by a poller or event may be needed by another poller or event that
1036 * runs on the same pthread. This can lead to deadlock or detection of deadlock.
1037 * - A lock that is acquired by a poller or event that is needed by another poller or event that
1038 * runs on a second pthread will block the second pthread from doing any useful work until the
1039 * lock is released. Because the lock holder and the lock acquirer are on the same pthread, this
1040 * would lead to deadlock.
1041 *
1042 * If an SPDK spinlock is used erroneously, the program will abort.
1043 */
1044struct spdk_spinlock {
1045 pthread_spinlock_t spinlock;
1046 struct spdk_thread *thread;
1047 struct spdk_spinlock_internal *internal;
1048 bool initialized;
1049 bool destroyed;
1050};
