127/**
128 * Add callback called when connection is closed. Pair of \c cb and \c ctx must be unique or error is returned.
129 * Registered callback is called only once and there is no need to call \c spdk_jsonrpc_conn_del_close_cb
130 * inside from \c cb.
131 *
132 * \note Current implementation allow only one close callback per connection.
133 *
134 * \param conn JSON RPC server connection
135 * \param cb callback function
136 * \param ctx argument for \c cb
137 *
138 * \return 0 on success, or negated errno code:
139 * -EEXIST \c cb and \c ctx is already registered
140 * -ENOTCONN Callback can't be added because connection is closed.
141 * -ENOSPC no more space to register callback.
142 */
143int spdk_jsonrpc_conn_add_close_cb(struct spdk_jsonrpc_server_conn *conn,
144 spdk_jsonrpc_conn_closed_fn cb, void *ctx);
145

214/**
215 * Begin building a JSON-RPC request.
216 *
217 * If this function returns non-NULL, the user must call spdk_jsonrpc_end_request()
218 * on the request after writing the desired request object to the spdk_json_write_ctx.
219 *
220 * This function works for both single requests and batch requests. If
221 * spdk_jsonrpc_begin_batch() was called first, this adds a request to the batch
222 * and returns the same write context. In batch mode, if id < 0, an ID is
223 * auto-assigned starting from 0.
224 *
225 * \param request JSON-RPC request.
226 * \param id ID index for the request. If < 0, skip ID (single) or auto-assign (batch).
227 * \param method Name of the RPC method. If NULL caller will have to create "method" key.
228 *
229 * \return JSON write context or NULL in case of error.
230 */
231struct spdk_json_write_ctx *
232spdk_jsonrpc_begin_request(struct spdk_jsonrpc_client_request *request, int32_t id,
233 const char *method);

306/**
307 * Send the JSON-RPC request in JSON-RPC client. Library takes ownership of the
308 * request object and will free it when done.
309 *
310 * This function is not thread safe and should only be called from one thread at
311 * a time while no other threads are actively \c client object.
312 *
313 * \param client JSON-RPC client.
314 * \param req JSON-RPC request.
315 *
316 * \return 0 on success or negative error code.
317 * -ENOSPC - no space left to queue another request. Try again later.
318 */
319int spdk_jsonrpc_client_send_request(struct spdk_jsonrpc_client *client,
320 struct spdk_jsonrpc_client_request *req);
321
322/**
323 * Poll the JSON-RPC client. When any response is available use
324 * \c spdk_jsonrpc_client_get_response to retrieve it.
325 *
326 * This function is not thread safe and should only be called from one thread at
327 * a time while no other threads are actively \c client object.
328 *
329 * \param client JSON-RPC client.
330 * \param timeout Time in milliseconds this function will block. -1 block forever, 0 don't block.
331 *
332 * \return If no error occurred, this function returns a non-negative number indicating how
333 * many ready responses can be retrieved. If an error occurred, this function returns one of
334 * the following negated errno values:
335 * -ENOTCONN - not connected yet. Try again later.
336 * -EINVAL - response is detected to be invalid. Client connection should be terminated.
337 * -ENOSPC - no space to receive another response. User need to retrieve waiting responses.
338 * -EIO - connection terminated (or other critical error). Client connection should be terminated.
339 * -ENOMEM - out of memory
340 */
341int spdk_jsonrpc_client_poll(struct spdk_jsonrpc_client *client, int timeout);
342
