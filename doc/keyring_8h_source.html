15struct spdk_key;
16
17/**
18 * Get a reference to a key from the keyring. The key must have been added to the keyring by
19 * the appropriate keyring module. The reference will be kept alive until its released via
20 * `spdk_keyring_put_key()`. If the key is removed from the keyring, the reference is kept alive, but
21 * the key won't be usable anymore.
22 *
23 * \param name Name of a key. The name can be optionally prepended with the name of a keyring to
24 * retrieve the key from followed by a ":" character. For instance, "keyring0:key0" would retrieve
25 * a key "key0" from keyring "keyring0". If omitted, "global" keyring will be used. To get a key
26 * with a ":" character in its name from the global keyring, empty keyring name should be specified
27 * (e.g. ":key0:foo" refers to a key "key0:foo" in the global keyring).
28 *
29 * \return Reference to a key or NULL if the key doesn't exist.
30 */
31struct spdk_key *spdk_keyring_get_key(const char *name);
32
33/**
34 * Release a reference to a key obtained from `spdk_keyring_get_key()`.
35 *
36 * \param key Reference to a key. If NULL, this function is a no-op.
37 */
38void spdk_keyring_put_key(struct spdk_key *key);
39
40/**
41 * Get the name of a key.
42 *
43 * \param key Reference to a key.
44 *
45 * \return Name of the key.
46 */
47const char *spdk_key_get_name(struct spdk_key *key);
48
49/**
50 * Retrieve keying material from a key reference.
51 *
52 * \param key Reference to a key.
53 * \param buf Buffer to write the data to.
54 * \param len Size of the `buf` buffer.
55 *
56 * \return The number of bytes written to `buf` or negative errno on error.
57 */
58int spdk_key_get_key(struct spdk_key *key, void *buf, int len);
59
60/**
61 * Duplicate a key. The returned key reference might be a pointer to the same exact object. After
62 * duplicating a key, the new reference should be released via `spdk_keyring_put_key()`.
63 *
64 * \param key Reference to a key.
65 *
66 * \return Pointer to the key reference.
67 */
68struct spdk_key *spdk_key_dup(struct spdk_key *key);
69
70/**
71 * Initialize the keyring library.
72 *
73 * \return 0 on success, negative errno otherwise.
74 */
75int spdk_keyring_init(void);
76
77/**
78 * Free any resources acquired by the keyring library. This function will free all of the keys.
79 */
80void spdk_keyring_cleanup(void);

82struct spdk_keyring;

84/** Iterate over all keys including those that were removed, but still have active references */
85#define SPDK_KEYRING_FOR_EACH_ALL 0x1

87/**
88 * Execute a function on each registered key attached to a given keyring. For now, this function
89 * only supports iterating over keys from all keyrings and the `keyring` parameter must be set to
90 * NULL.
91 *
92 * \param keyring Keyring over which to iterate. If NULL, iterate over keys from all keyrings.
93 * \param ctx Context to pass to the function.
94 * \param fn Function to call.
95 * \param flags Flags controlling the keys to iterate over.
96 */
97void spdk_keyring_for_each_key(struct spdk_keyring *keyring, void *ctx,
98 void (*fn)(void *ctx, struct spdk_key *key), uint32_t flags);
