253static inline uint64_t
254spdk_trace_file_get_sections_size(const struct spdk_trace_file *f)
255{
256 struct spdk_trace_section_owner *os = spdk_trace_get_owner_section(f);
257 struct spdk_trace_section_tpoint_mask *tpm = spdk_trace_get_tpoint_mask_section(f);
258 struct spdk_trace_section_owner_type *ot = spdk_trace_get_owner_type_section(f);
259 struct spdk_trace_section_object *obj = spdk_trace_get_object_section(f);
260 struct spdk_trace_section_tpoint *tp = spdk_trace_get_tpoint_section(f);
261 struct spdk_trace_section_lcore_offsets *lc = spdk_trace_get_lcore_offsets_section(f);
262
263 return sizeof(struct spdk_trace_section_main) +
264 sizeof(struct spdk_trace_section_owner) +
265 os->num_owners * (sizeof(struct spdk_trace_owner) + os->owner_description_size) +
266 sizeof(struct spdk_trace_section_tpoint_mask) +
267 tpm->count * sizeof(uint64_t) +
268 sizeof(struct spdk_trace_section_owner_type) +
269 ot->count * sizeof(struct spdk_trace_owner_type) +
270 sizeof(struct spdk_trace_section_object) +
271 obj->count * sizeof(struct spdk_trace_object) +
272 sizeof(struct spdk_trace_section_tpoint) +
273 tp->count * sizeof(struct spdk_trace_tpoint) +
274 sizeof(struct spdk_trace_section_lcore_offsets) +
275 lc->count * sizeof(uint64_t);
276}

278static inline struct spdk_trace_history *
279spdk_get_per_lcore_history(struct spdk_trace_file *trace_file, unsigned lcore)
280{
281 struct spdk_trace_section_lcore_offsets *section;
282 uint64_t lcore_history_offset;
283
284 section = spdk_trace_get_lcore_offsets_section(trace_file);
285 if (section == NULL || lcore >= section->count) {
286 return NULL;
287 }
288
289 lcore_history_offset = section->lcore_offsets[lcore];
290 if (lcore_history_offset == 0) {
291 return NULL;
292 }
293
294 return (struct spdk_trace_history *)(((char *)trace_file) + lcore_history_offset);
295}

297static inline struct spdk_trace_owner *
298spdk_get_trace_owner(const struct spdk_trace_file *trace_file, uint16_t owner_id)
299{
300 struct spdk_trace_section_owner *section;
301 uint64_t owner_size;
302
303 section = spdk_trace_get_owner_section(trace_file);
304 if (section == NULL || owner_id >= section->num_owners) {
305 return NULL;
306 }
307
308 owner_size = sizeof(struct spdk_trace_owner) + section->owner_description_size;
309 return (struct spdk_trace_owner *)(section->data + owner_id * owner_size);
310}
