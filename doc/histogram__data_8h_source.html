20#define SPDK_HISTOGRAM_GRANULARITY_DEFAULT 7
21#define SPDK_HISTOGRAM_GRANULARITY(h) h->granularity
22#define SPDK_HISTOGRAM_BUCKET_LSB(h) (64 - SPDK_HISTOGRAM_GRANULARITY(h))
23#define SPDK_HISTOGRAM_NUM_BUCKETS_PER_RANGE(h) (1ULL << SPDK_HISTOGRAM_GRANULARITY(h))
24#define SPDK_HISTOGRAM_BUCKET_MASK(h) (SPDK_HISTOGRAM_NUM_BUCKETS_PER_RANGE(h) - 1)
25#define SPDK_HISTOGRAM_NUM_BUCKET_RANGES(h) (h->max_range - h->min_range + 1)
26#define SPDK_HISTOGRAM_NUM_BUCKETS(h) (SPDK_HISTOGRAM_NUM_BUCKETS_PER_RANGE(h) * \
27 SPDK_HISTOGRAM_NUM_BUCKET_RANGES(h))

29/*
30 * SPDK histograms are implemented using ranges of bucket arrays. The most common usage
31 * model is using TSC datapoints to capture an I/O latency histogram. For this usage model,
32 * the histogram tracks only TSC deltas - any translation to microseconds is done by the
33 * histogram user calling spdk_histogram_data_iterate() to iterate over the buckets to perform
34 * the translations.
35 *
36 * Each range has a number of buckets determined by SPDK_HISTOGRAM_NUM_BUCKETS_PER_RANGE
37 * which is 128. The buckets in ranges 0 and 1 each map to one specific datapoint value.
38 * The buckets in subsequent ranges each map to twice as many datapoint values as buckets
39 * in the range before it:
40 *
41 * Range 0: 1 value each - 128 buckets cover 0 to 127 (2^7-1)
42 * Range 1: 1 value each - 128 buckets cover 128 to 255 (2^8-1)
43 * Range 2: 2 values each - 128 buckets cover 256 to 511 (2^9-1)
44 * Range 3: 4 values each - 128 buckets cover 512 to 1023 (2^10-1)
45 * Range 4: 8 values each - 128 buckets cover 1024 to 2047 (2^11-1)
46 * Range 5: 16 values each - 128 buckets cover 2048 to 4095 (2^12-1)
47 * ...
48 * Range 55: 2^54 values each - 128 buckets cover 2^61 to 2^62-1
49 * Range 56: 2^55 values each - 128 buckets cover 2^62 to 2^63-1
50 * Range 57: 2^56 values each - 128 buckets cover 2^63 to 2^64-1
51 *
52 * On a 2.3GHz processor, this strategy results in 50ns buckets in the 7-14us range (sweet
53 * spot for Intel Optane SSD latency testing).
54 *
55 * Buckets can be made more granular by increasing SPDK_HISTOGRAM_GRANULARITY. This
56 * comes at the cost of additional storage per namespace context to store the bucket data.
57 * In order to lower number of ranges to shrink unnecessary low and high datapoints
58 * min_val and max_val can be specified with spdk_histogram_data_alloc_sized_ext().
59 * It will limit the values in histogram to a range [min_val, max_val).
60 */
61
62struct spdk_histogram_data {
63 uint32_t granularity;
64 uint32_t min_range;
65 uint32_t max_range;
66 uint64_t *bucket;
67};

69static inline void
70__spdk_histogram_increment(struct spdk_histogram_data *h, uint32_t range, uint32_t index)
71{
72 uint64_t *count;
73
74 count = &h->bucket[((range - h->min_range) << SPDK_HISTOGRAM_GRANULARITY(h)) + index];
75 (*count)++;
76}

78static inline uint64_t
79__spdk_histogram_get_count(const struct spdk_histogram_data *h, uint32_t range, uint32_t index)
80{
81 return h->bucket[((range - h->min_range) << SPDK_HISTOGRAM_GRANULARITY(h)) + index];
82}

85__spdk_histogram_get_bucket(const struct spdk_histogram_data *h, uint32_t range, uint32_t index)
86{
87 return &h->bucket[((range - h->min_range) << SPDK_HISTOGRAM_GRANULARITY(h)) + index];

96static inline uint32_t
97__spdk_histogram_data_get_bucket_range(struct spdk_histogram_data *h, uint64_t datapoint)
98{
99 uint32_t clz, range;
100
101 clz = datapoint > 0 ? __builtin_clzll(datapoint) : 64;
102
103 if (clz <= SPDK_HISTOGRAM_BUCKET_LSB(h)) {
104 range = SPDK_HISTOGRAM_BUCKET_LSB(h) - clz;
105 } else {
106 range = 0;
107 }
108
109 return range;
110}

112static inline uint32_t
113__spdk_histogram_data_get_bucket_index(struct spdk_histogram_data *h, uint64_t datapoint,
114 uint32_t range)
115{
116 uint32_t shift;
117
118 if (range == 0) {
119 shift = 0;
120 } else {
121 shift = range - 1;
122 }
123
124 return (datapoint >> shift) & SPDK_HISTOGRAM_BUCKET_MASK(h);
125}

127static inline void
128spdk_histogram_data_tally(struct spdk_histogram_data *histogram, uint64_t datapoint)
129{
130 uint32_t range, index;
131
132 range = __spdk_histogram_data_get_bucket_range(histogram, datapoint);
133
134 if (range < histogram->min_range) {
135 range = histogram->min_range;
136 index = 0;
137 } else if (range > histogram->max_range) {
138 range = histogram->max_range;
139 index = SPDK_HISTOGRAM_NUM_BUCKETS_PER_RANGE(histogram) - 1;
140 } else {
141 index = __spdk_histogram_data_get_bucket_index(histogram, datapoint, range);
142 }
143
144 __spdk_histogram_increment(histogram, range, index);
145}

147static inline uint64_t
148__spdk_histogram_data_get_bucket_start(const struct spdk_histogram_data *h, uint32_t range,
149 uint32_t index)
150{
151 uint64_t bucket;
152
153 index += 1;
154 if (range > 0) {
155 bucket = 1ULL << (range + SPDK_HISTOGRAM_GRANULARITY(h) - 1);
156 bucket += (uint64_t)index << (range - 1);
157 } else {
158 bucket = index;
159 }
160
161 return bucket;
162}

164typedef void (*spdk_histogram_data_fn)(void *ctx, uint64_t start, uint64_t end, uint64_t count,
165 uint64_t total, uint64_t so_far);
166
167static inline void
168spdk_histogram_data_iterate(const struct spdk_histogram_data *histogram,
169 spdk_histogram_data_fn fn, void *ctx)
170{
171 uint64_t i, j, count, so_far, total;
172 uint64_t bucket, last_bucket;
173
174 total = 0;
175
176 for (i = histogram->min_range; i <= histogram->max_range; i++) {
177 for (j = 0; j < SPDK_HISTOGRAM_NUM_BUCKETS_PER_RANGE(histogram); j++) {
178 total += __spdk_histogram_get_count(histogram, i, j);
179 }
180 }
181
182 so_far = 0;
183 bucket = 0;
184
185 for (i = histogram->min_range; i <= histogram->max_range; i++) {
186 for (j = 0; j < SPDK_HISTOGRAM_NUM_BUCKETS_PER_RANGE(histogram); j++) {
187 count = __spdk_histogram_get_count(histogram, i, j);
188 so_far += count;
189 last_bucket = bucket;
190 bucket = __spdk_histogram_data_get_bucket_start(histogram, i, j);
191 fn(ctx, last_bucket, bucket, count, total, so_far);
192 }
193 }
194}

196static inline int
197spdk_histogram_data_merge(const struct spdk_histogram_data *dst,
198 const struct spdk_histogram_data *src)
199{
200 uint64_t i;
201
202 /* Histograms with different granularity values cannot be simply
203 * merged, because the buckets represent different ranges of
204 * values.
205 */
206 if (dst->granularity != src->granularity) {
207 return -EINVAL;
208 }
209
210 /* Histogram with different size cannot be simply merged. */
211 if (dst->min_range != src->min_range || dst->max_range != src->max_range) {
212 return -EINVAL;
213 }
214
215 for (i = 0; i < SPDK_HISTOGRAM_NUM_BUCKETS(dst); i++) {
216 dst->bucket[i] += src->bucket[i];
217 }
218
219 return 0;
220}

222/**
223 * Allocate a histogram data structure with specified granularity. It tracks datapoints
224 * from min_val (inclusive) to max_val (exclusive).
225 *
226 * \param granularity Granularity of the histogram buckets. Each power-of-2 range is
227 * split into (1 << granularity) buckets.
228 * \param min_val The minimum value to be tracked, inclusive.
229 * \param max_val The maximum value to be tracked, exclusive.
230 *
231 * \return A histogram data structure.
232 */
233static inline struct spdk_histogram_data *
234spdk_histogram_data_alloc_sized_ext(uint32_t granularity, uint64_t min_val, uint64_t max_val)
235{
236 struct spdk_histogram_data *h;
237
238 if (min_val >= max_val) {
239 return NULL;
240 }
241
242 h = (struct spdk_histogram_data *)calloc(1, sizeof(*h));
243 if (h == NULL) {
244 return NULL;
245 }
246
247 h->granularity = granularity;
248 h->min_range = __spdk_histogram_data_get_bucket_range(h, min_val);
249 h->max_range = __spdk_histogram_data_get_bucket_range(h, max_val - 1);
250 h->bucket = (uint64_t *)calloc(SPDK_HISTOGRAM_NUM_BUCKETS(h), sizeof(uint64_t));
251 if (h->bucket == NULL) {
252 free(h);
253 return NULL;
254 }
255
256 return h;
257}

259static inline struct spdk_histogram_data *
260spdk_histogram_data_alloc_sized(uint32_t granularity)
261{
262 return spdk_histogram_data_alloc_sized_ext(granularity, 0, UINT64_MAX);
263}

289static inline void
290__spdk_histogram_get_borders_count_cb(void *_ctx, uint64_t start, uint64_t end,
291 uint64_t count,
292 uint64_t total, uint64_t so_far)
293{
294 struct spdk_histogram_borders *ctx = (struct spdk_histogram_borders *)_ctx;
295 size_t old_index;
296
297 while ((ctx->index < ctx->num_borders_count - 1) && start > ctx->borders[ctx->index]) {
298 old_index = ctx->index;
299 ctx->index++;
300 ctx->border_count[ctx->index] = ctx->border_count[old_index];
301 }
302
303 ctx->border_count[ctx->index] = so_far;
304}

319/**
320 * Allocate a context to dump histogram buckets for specified borders.
321 * The borders array must be sorted in ascending order, each border represents a value which is an upper bound of a bucket.
322 * border_count will contain the count of all values that are less or equal to its border value.
323 * Reported border_count will contain one more element than the borders array, the last element will represent total count of all elements in the histogram.
324 *
325 * \param borders An array of border values, must be sorted in ascending order.
326 * \param num_borders The number of borders in the borders array.
327 *
328 * \return A context to dump histogram buckets for specified borders.
329 */
330static inline struct spdk_histogram_borders *
331spdk_histogram_get_borders_count_alloc_ctx(uint64_t *borders, size_t num_borders)
332{
333 struct spdk_histogram_borders *ctx;
334
335 if (num_borders == 0) {
336 return NULL;
337 }
338
339 ctx = (struct spdk_histogram_borders *)calloc(1,
340 sizeof(*ctx) + sizeof(uint64_t) * (num_borders + 1));
341 if (!ctx) {
342 return NULL;
343 }
344
345 ctx->borders = borders;
346 /** +1 for bucket containing values from last border to +Inf */
347 ctx->num_borders_count = num_borders + 1;
348
349 return ctx;
350}
