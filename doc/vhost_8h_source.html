92/**
93 * SPDK vhost device (vdev). An equivalent of Virtio device.
94 * Both virtio-blk and virtio-scsi devices are represented by this
95 * struct. For virtio-scsi a single vhost device (also called SCSI
96 * controller) may contain multiple SCSI targets (devices), each of
97 * which may contain multiple logical units (SCSI LUNs). For now
98 * only one LUN per target is available.
99 *
100 * All vdev-changing functions operate directly on this object.
101 * Note that \c spdk_vhost_dev cannot be acquired. This object is
102 * only accessible as a callback parameter via \c
103 * spdk_vhost_call_external_event and it's derivatives. This ensures
104 * that all access to the vdev is piped through a single,
105 * thread-safe API.
106 */
107struct spdk_vhost_dev;

174/**
175 * By default, events are generated when asked, but for high queue depth and
176 * high IOPS this prove to be inefficient both for guest kernel that have to
177 * handle a lot more IO completions and for SPDK vhost that need to make more
178 * syscalls. If enabled, limit amount of events (IRQs) sent to initiator by SPDK
179 * vhost effectively coalescing couple of completions. This of course introduce
180 * IO latency penalty proportional to event delay time.
181 *
182 * Actual events delay time when is calculated according to below formula:
183 * if (delay_base == 0 || IOPS < iops_threshold) {
184 * delay = 0;
185 * } else if (IOPS < iops_threshold) {
186 * delay = delay_base * (iops - iops_threshold) / iops_threshold;
187 * }
188 *
189 * \param vdev vhost device.
190 * \param delay_base_us Base delay time in microseconds. If 0, coalescing is disabled.
191 * \param iops_threshold IOPS threshold when coalescing is activated.
192 */
193int spdk_vhost_set_coalescing(struct spdk_vhost_dev *vdev, uint32_t delay_base_us,
194 uint32_t iops_threshold);
195
196/**
197 * Get coalescing parameters.
198 *
199 * \see spdk_vhost_set_coalescing
200 *
201 * \param vdev vhost device.
202 * \param delay_base_us Optional pointer to store base delay time.
203 * \param iops_threshold Optional pointer to store IOPS threshold.
204 */
205void spdk_vhost_get_coalescing(struct spdk_vhost_dev *vdev, uint32_t *delay_base_us,
206 uint32_t *iops_threshold);

208/**
209 * Construct an empty vhost SCSI device. This will create a
210 * Unix domain socket together with a vhost-user slave server waiting
211 * for a connection on this socket. Creating the vdev does not
212 * start any I/O pollers and does not hog the CPU. I/O processing
213 * starts after receiving proper message on the created socket.
214 * See QEMU's vhost-user documentation for details.
215 * All physical devices have to be separately attached to this
216 * vdev via \c spdk_vhost_scsi_dev_add_tgt().
217 *
218 * This function is thread-safe.
219 *
220 * \param name name of the vhost device. The name will also be used
221 * for socket name, which is exactly \c socket_base_dir/name
222 * \param cpumask string containing cpumask in hex. The leading *0x*
223 * is allowed but not required. The mask itself can be constructed as:
224 * ((1 << cpu0) | (1 << cpu1) | ... | (1 << cpuN)).
225 *
226 * \return 0 on success, negative errno on error.
227 */
228int spdk_vhost_scsi_dev_construct(const char *name, const char *cpumask);

246/**
247 * Construct and attach new SCSI target to the vhost SCSI device
248 * on given (unoccupied) slot. The device will be created with a single
249 * LUN0 associated with given SPDK bdev. Currently only one LUN per
250 * device is supported.
251 *
252 * If the vhost SCSI device has an active connection and has negotiated
253 * \c VIRTIO_SCSI_F_HOTPLUG feature, the new SCSI target should be
254 * automatically detected by the other side.
255 *
256 * \param vdev vhost SCSI device.
257 * \param scsi_tgt_num slot to attach to or negative value to use first free.
258 * \param bdev_name name of the SPDK bdev to associate with SCSI LUN0.
259 *
260 * \return value >= 0 on success - the SCSI target ID, negative errno code:
261 * -EINVAL - one of the arguments is invalid:
262 * - vdev is not vhost SCSI device
263 * - SCSI target ID is out of range
264 * - bdev name is NULL
265 * - can't create SCSI LUN because of other errors e.g.: bdev does not exist
266 * -ENOSPC - scsi_tgt_num is -1 and maximum targets in vhost SCSI device reached
267 * -EEXIST - SCSI target ID already exists
268 */
269int spdk_vhost_scsi_dev_add_tgt(struct spdk_vhost_dev *vdev, int scsi_tgt_num,
270 const char *bdev_name);

303/**
304 * Construct a vhost blk device. This will create a Unix domain
305 * socket together with a vhost-user slave server waiting for a
306 * connection on this socket. Creating the vdev does not start
307 * any I/O pollers and does not hog the CPU. I/O processing starts
308 * after receiving proper message on the created socket.
309 * See QEMU's vhost-user documentation for details. Vhost blk
310 * device is tightly associated with given SPDK bdev. Given
311 * bdev can not be changed, unless it has been hotremoved. This
312 * would result in all I/O failing with virtio \c VIRTIO_BLK_S_IOERR
313 * error code.
314 *
315 * This function is thread-safe.
316 *
317 * \param name name of the vhost blk device. The name will also be
318 * used for socket name, which is exactly \c socket_base_dir/name
319 * \param cpumask string containing cpumask in hex. The leading *0x*
320 * is allowed but not required. The mask itself can be constructed as:
321 * ((1 << cpu0) | (1 << cpu1) | ... | (1 << cpuN)).
322 * \param dev_name bdev name to associate with this vhost device
323 * \param transport virtio blk transport name (default: vhost_user_blk)
324 * \param params JSON value object containing variables:
325 * readonly if set, all writes to the device will fail with
326 * \c VIRTIO_BLK_S_IOERR error code.
327 * packed_ring this controller supports packed ring if set.
328 *
329 * \return 0 on success, negative errno on error.
330 */
331int spdk_vhost_blk_construct(const char *name, const char *cpumask, const char *dev_name,
332 const char *transport, const struct spdk_json_val *params);
